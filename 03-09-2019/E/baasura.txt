#include <bits/stdc++.h>
#include <iostream>

using namespace std;

char mapa[400][400];
int R = 0, C = 0;
int cont = 0;

void Mueve(int r, int c, int ccc){
    if ((r == R || c == C) && mapa[c][r] == 'D') {
        cont = ccc;
    }
    else {
        if (r + 1 < R && mapa[c][r+1] != '#' && mapa[c][r+1] != 'l') {
            if (mapa[c][r] == 'c') {
                mapa[c][r] = 'l';
                Mueve(c, r+1, ccc+1);
                mapa[c][r] = 'c';
            }
            else {
                mapa[c][r] = 'l';
                Mueve(c, r + 1, ccc);
                mapa[c][r] = 'c';
            }
        }
        else {
            if (r - 1 < R && mapa[c][r-1] != '#' && mapa[c][r-1] != 'l') {
                if (mapa[c][r] == 'c') {
                    mapa[c][r] = 'l';
                    Mueve(c, r-1, ccc+1);
                    mapa[c][r] = 'c';
                }
                else {
                    mapa[c][r] = 'l';
                    Mueve(c, r - 1, ccc);
                    mapa[c][r] = 'c';
                }
            }
            else {
                if (c + 1 < C && mapa[c + 1][r] != '#' && mapa[c + 1][r] != 'l') {
                    if (mapa[c][r] == 'c') {
                        mapa[c][r] = 'l';
                        Mueve(c + 1, r, ccc+1);
                        mapa[c][r] = 'c';
                    }
                    else {
                        mapa[c][r] = 'l';
                        Mueve(c + 1, r, ccc);
                        mapa[c][r] = 'c';
                    }
                }

                else {
                    if (c - 1 < C && mapa[c - 1][r] != '#' && mapa[c + 1][r] != 'l') {
                        if (mapa[c][r] == 'c') {
                            mapa[c][r] = 'l';
                            Mueve(c - 1, r, ccc+1);
                            mapa[c][r] = 'c';
                        }
                        else {
                            mapa[c][r] = 'l';
                            Mueve(c - 1, r, ccc);
                            mapa[c][r] = 'c';
                        }
                    }
                }
            };
        }
    }
}

int main() {
    int r, c;
    cin >> R >> C;
    for (int i = 0; i < R; i++) {
        for (int j = 0; j < C; j++) {
            char t;
            cin >> t ;
            mapa[i][j] = t;
        }
    }

    cin >> r >> c;

    int resp = 0;
    Mueve(c-1, r-1, 0);
    cout << resp;
    return 0;
}
